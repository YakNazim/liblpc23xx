00058 /*
00059  * __set_cpsr
00060  * ---------------------------------------
00061  * Set the value of the cpsr
00062  * This will force the processor into, then out of system (privileged) mode.
00063  *  Resulting processor mode is decided by caller by setting mode bits in val.
00064  * Don't change the 'T' bit.
00065  *  "Never use an MSR instruction to force a change to the state of the T bit in the CPSR. If
00066  *  you do this, the processor enters an unpredictable state."
00067  *
00068  * This is a very dangerous function, use at your own risk.
00069  * Reference: ARM7TDMI-S Technical Reference Manual, p54 "The Control Bits"
00070  */
00071 static inline void __set_cpsr(uint32_t val) {
00072     uint32_t current_cpsr ;
00073     uint32_t save_Tbit;
00074    
00075     current_cpsr = __get_cpsr();
00076 
00077     save_Tbit    = (current_cpsr & (1<<T_BIT));
00078     val          = (val & T_MASK) | save_Tbit;
00079 
00080     asm volatile (" msr  cpsr, %0" : /* no outputs */ : "r" (val)  );
00081 }
