00511 void USBHwISR(void)
00512 {
00513     uint32_t dwStatus;
00514     uint32_t dwIntBit;
00515     uint8_t  bEPStat, bDevStat, bStat;
00516     int i;
00517     uint16_t wFrame;
00518 
00519 // LED9 monitors total time in interrupt routine
00520 DEBUG_LED_ON(9);
00521 
00522     // handle device interrupts
00523     dwStatus = USBDevIntSt;
00524     
00525     // frame interrupt
00526     if (dwStatus & FRAME) {
00527         // clear int
00528         USBDevIntClr = FRAME;
00529         // call handler
00530         if (_pfnFrameHandler != NULL) {
00531             wFrame = USBHwCmdRead(CMD_DEV_READ_CUR_FRAME_NR);
00532             _pfnFrameHandler(wFrame);
00533         }
00534     }
00535     
00536     // device status interrupt
00537     if (dwStatus & DEV_STAT) {
00538         /*  Clear DEV_STAT interrupt before reading DEV_STAT register.
00539             This prevents corrupted device status reads, see
00540             LPC2148 User manual revision 2, 25 july 2006.
00541         */
00542         USBDevIntClr = DEV_STAT;
00543         bDevStat = USBHwCmdRead(CMD_DEV_STATUS);
00544         if (bDevStat & (CON_CH | SUS_CH | RST)) {
00545             // convert device status into something HW independent
00546             bStat = ((bDevStat & CON) ? DEV_STATUS_CONNECT : 0) |
00547                     ((bDevStat & SUS) ? DEV_STATUS_SUSPEND : 0) |
00548                     ((bDevStat & RST) ? DEV_STATUS_RESET : 0);
00549             // call handler
00550             if (_pfnDevIntHandler != NULL) {
00551 DEBUG_LED_ON(8);        
00552                 _pfnDevIntHandler(bStat);
00553 DEBUG_LED_OFF(8);       
00554             }
00555         }
00556     }
00557     
00558     // endpoint interrupt
00559     if (dwStatus & EP_SLOW) {
00560         // clear EP_SLOW
00561         USBDevIntClr = EP_SLOW;
00562         // check all endpoints
00563         for (i = 0; i < 32; i++) {
00564             dwIntBit = (1 << i);
00565             if (USBEpIntSt & dwIntBit) {
00566                 // clear int (and retrieve status)
00567                 USBEpIntClr = dwIntBit;
00568                 Wait4DevInt(CDFULL);
00569                 bEPStat = USBCmdData;
00570                 // convert EP pipe stat into something HW independent
00571                 bStat = ((bEPStat & EPSTAT_FE) ? EP_STATUS_DATA : 0) |
00572                         ((bEPStat & EPSTAT_ST) ? EP_STATUS_STALLED : 0) |
00573                         ((bEPStat & EPSTAT_STP) ? EP_STATUS_SETUP : 0) |
00574                         ((bEPStat & EPSTAT_EPN) ? EP_STATUS_NACKED : 0) |
00575                         ((bEPStat & EPSTAT_PO) ? EP_STATUS_ERROR : 0);
00576                 // call handler
00577                 if (_apfnEPIntHandlers[i / 2] != NULL) {
00578 DEBUG_LED_ON(10);       
00579                     _apfnEPIntHandlers[i / 2](IDX2EP(i), bStat);
00580 DEBUG_LED_OFF(10);
00581                 }
00582             }
00583         }
00584     }
00585     
00586 DEBUG_LED_OFF(9);
